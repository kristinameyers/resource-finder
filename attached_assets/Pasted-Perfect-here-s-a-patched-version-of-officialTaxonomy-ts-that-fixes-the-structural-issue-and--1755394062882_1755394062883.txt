/**
 * Get the official taxonomy code for a main category
 */
export function getOfficialCategoryTaxonomyCode(categoryId: string): string | null {
  const key = categoryId.toLowerCase();
  const category = MAIN_CATEGORIES[key as keyof typeof MAIN_CATEGORIES];
  if (!category) {
    console.warn(`[Taxonomy] Unknown main categoryId="${categoryId}"`);
    return null;
  }
  return category.taxonomyCode;
}

/**
 * Get the official taxonomy code for a subcategory
 */
export function getOfficialSubcategoryTaxonomyCode(categoryId: string, subcategoryId: string): string | null {
  const key = categoryId.toLowerCase();
  const subcategories = SUBCATEGORIES[key as keyof typeof SUBCATEGORIES];
  if (!subcategories) {
    console.warn(`[Taxonomy] No subcategories found for categoryId="${categoryId}"`);
    return null;
  }
  const subcategory = subcategories.find(sub => sub.id === subcategoryId);
  if (!subcategory) {
    console.warn(`[Taxonomy] Unknown subcategoryId="${subcategoryId}" under category="${categoryId}"`);
    return null;
  }
  return subcategory.taxonomyCode;
}
